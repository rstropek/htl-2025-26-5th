        int Id PK
        string Name "max 100 Zeichen"
        int LaufkategorieId FK
        decimal Streckenlänge "min 0.01, 2 Dezimalstellen"
        date Datum
        string Ort "max 100 Zeichen"
    }

    Teilnehmer {
        int Id PK
        int LaufbewerbId FK
        int Startnummer "min 1, eindeutig je Bewerb"

        string Nachname "max 100 Zeichen"
        int AngestrebteGesamtzeit "Sekunden, min 1"
    }
